\section*{Basic Problems}
\begin{enumerate}
    \item Determine $(8 \cdot 16) \pmod{7}$ \bspace
    \item Determine the remainder when $124 \cdot 134 \cdot 23 \cdot 49 \cdot 235 \cdot 13$ is divided by $3$. \bspace
    \item Determine the remainder when $124 + 234 + 43 + 56 + 23 + 12 + 78$ is divided by $3$. \bspace
    \item Show that if $a - b$ and $a + b$ is divisible by $n$, that $a^2 - b^2$ is divisible by $n$. \bspace
\end{enumerate}

\section*{Intermediate Problems}
\begin{enumerate}
    \item Show that a square is always in the $4k$ or $4k + 1$. \ispace
    \item Determine all integers $a$ such that $\displaystyle{\frac{5}{a + 2}}$ is an integer. \ispace
    \item Suppose that $n$ is a positive integer and $a = \displaystyle{\frac{10^{2n} - 1}{3(10^n - 1)}}$. If the sum of the digits of $a$ is $567$, determine the value of $n$. \ispace
\end{enumerate}

\section*{Advanced Problems}
\begin{enumerate}
    \item Show that if $p > 3$ is prime, that $p^2 \equiv 1 \pmod{24}$ \aspace
    \item Show that $n^2 + 23$ is divisible by $24$ for infinitely many $n$. \aspace
\end{enumerate}
