\section*{Basic Problems}
\begin{enumerate}
    \item Prove by induction that $1^2 + 2^2 + 3^2 + ... + n^2 = \frac{n(n+1)(2n+1)}{6}$ for every positive integer $n$. \bspace
    \item Show that for all positive integers $n$, $7 + 6 \cdot 7 + 6 \cdot 7^2 + ... + 6 \cdot 7^n = 7^{n+1}$. \bspace
    \item Prove the following for all positive integers: $$\frac{1}{1 \cdot 2} + \frac{1}{2 \cdot 3} + ... + \frac{1}{n(n+1)} = \frac{n}{n+1}$$ \bspace
\end{enumerate}

\section*{Intermediate Problems}
\begin{enumerate}
    \item Show that $(1 + 2 + 3 + ... + n)^2 = 1^3 + 2^3 + 3^3 + ... + n^3$. \ispace
    \item Prove that $n^3 - n$ is divisible by 6 for all positive integers. \ispace
\end{enumerate}

\section*{Advanced Problems}
\begin{enumerate}
    \item Prove that $2^{n+2} + 3^{2n+1}$ is divisible by 7. \aspace
    \item Prove the following for $n \ge 2$: $$1 + \frac{1}{\sqrt{2}} + \frac{1}{\sqrt{3}} + ... + \frac{1}{\sqrt{n}} < 2\sqrt{n} - 1$$ \aspace
    \item Prove that $a^{2^n} - 1$ is divisible by $4 \cdot 2^n$ for all odd integers $a$, and for all integers $n$.
\end{enumerate}
