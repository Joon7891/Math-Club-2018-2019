\section*{Basic Problems}
\begin{enumerate}
    \item Rationalize $\displaystyle{\frac{5 - 4i}{2 + 3i}}$ \bspace
    \item Determine the argument of $-3 + 4i$ \bspace
    \item Determine the complex modulus of $4 - 2i$ \bspace
\end{enumerate}


\section*{Intermediate Problems}
\begin{enumerate}
    \item Determine the sum of real values of $x$ and $y$ for which the following equation is satisfied: $$\frac{(1+i)x-2i}{3+i} + \frac{(2-3i)y+i}{3-i} = i$$ \ispace
    \item Express $\displaystyle{\frac{-3x}{1 - 5xi} + \frac{3i}{3 + i}}$ in the form $a + bi$ ($a, b \in \mathbb{R}$) \ispace
    \item Determine the following sum: $$\sum_{k = 1}^{200} i^k$$ \bspace
    \item If $p$ and $q$ are real numbers and $2+\sqrt{3}i$ is a root of $x^2+px+q=0$, what are the values of $p$ and $q$? \ispace
\end{enumerate}
\section*{Advanced Problems}
\begin{enumerate}
    \item Prove that if $a + bi$ ($b \neq 0$) is a root of $x^2 + px + q = 0$ and $a, b, p, q \in \mathbb{R}$, then $a - bi$ is also a root of the quadratic equation. \aspace
    \item Can two complex numbers, $\sin{x} + i\cos{2x}$ and $\cos{x} - i\sin{2x}$, by conjugates of each other? If so, what is the possible value of $x$? \aspace
\end{enumerate}
