\section*{Intermediate Problems}
\begin{enumerate}
    \item Determine all values of $x$ such that $x - \frac{4}{x} = -3$. Do not use guess and check. \interSpace
    
    \item The points $A(5, -8), B(9, -30),$ and $C(n, n)$ lie on the same line. Determine the value of $n$. \interSpace
    
    \item Given $x + y = 4$ and $xy = 12$, determine the value of $x^2 + 5xy + y^2$. \interSpace
\end{enumerate}

\section*{Advanced Problems}
\begin{enumerate}
    \item The functions $f$ and $g$ satisfy the following:
    \begin{align*}
        f(x) + g(x) & = 3x + 5 \\
        f(x) - g(x) &= 5x + 7
    \end{align*} for all values of $x$. Determine the value of $4f(2)g(2)$. \advSpace
    
    \item If $a(a - 1) = b(b - 1)$, what is $a$ in terms of $b$? \advSpace
    
    \item If $T = x^2 + \frac{1}{x^2}$, determine $a$ and $b$ such that $x^6 + \frac{1}{x^6} = T^3 + aT + b$. \advSpace
\end{enumerate}
