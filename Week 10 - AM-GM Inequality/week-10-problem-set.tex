\section*{Basic Problems}
\begin{enumerate}
    \item Determine the minimum value of $4x + \frac{9}{x}$, where $x > 0$. \bspace
    \item Determine the minimum value of $x^2 + y^2 + z^2$, where $x, y, z > 0$ and $xyz = 1$. \bspace
\end{enumerate}

\section*{Intermediate Problems}
\begin{enumerate}
    \item Determine the maximum value of $xy(72 - 3x - 4y)$, where $x, y > 0$. \ispace
    \item Let $a$ be a positive constant. Minimize $x^2 + \frac{a}{x}$, where $x > 0$. \ispace
\end{enumerate}

\section*{Advanced Problems}
\begin{enumerate}
    \item For all $a, b, c > 0$, prove that $$(a + b)(b + c)(c + a) \geq 8abc$$ \aspace
    \item For all integers $n > 1$, prove that $$\Big(\frac{1}{2} + \frac{2}{3} + \dots + \frac{n}{n+1}\Big)^n > \frac{n^n}{n + 1}$$ \aspace
\end{enumerate}
