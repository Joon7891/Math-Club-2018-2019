\section*{Basic Problems}
\begin{enumerate}
    \item Determine $x$ such that $\log_x2 + \log_x4 + \log_x8 = 1$. \bspace
    
    \item If $\displaystyle{x^3y^5 = 2^{11}  3^{13}}$ and $\displaystyle{\frac{x}{y^2} = \frac{1}{27}}$, solve for $x$ and $y$. \bspace
\end{enumerate}

\section*{Intermediate Problems}
\begin{enumerate}
    \item Determine the sum of the following series:
    \begin{equation*}
        \log_{10}\frac{3}{2} + \log_{10}\frac{4}{3} + \log_{10}\frac{5}{4} + \dots + \log_{10}\frac{200}{199}
    \end{equation*} \ispace
    
    \item Solve the equations for the point of intersection of the graphs of $y = \log_2(2x)$ and $y = \log_4(x)$. \ispace
\end{enumerate}

\section*{Advanced Problems}
\begin{enumerate}
    \item Prove that $a, b,$ and $c$ form a geometric sequence if and only if $\log_x{a}, \log_x{b},$ and $\log_x{c}$ form an arithmetic sequence. \aspace
    
    \item  $\displaystyle{2^{x + 3} + 2^x = 3^{y + 2} - 3^y$ and $x$ and $y}$ are integers, determine the values of $x$ and $y$. \aspace
\end{enumerate}
