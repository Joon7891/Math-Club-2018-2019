\section*{Basic Problems}
\begin{enumerate}
    \item Determine $\gcd{(2442, 17171)}$. \bspace
    \item Determine integers $x, y$ such that $16457x + 1638y = 7$. \bspace
    \item Determine the number of ways to make $\$2.00$ from only quarters and nickels. \bspace
\end{enumerate}

\section*{Intermediate Problems}
\begin{enumerate}
    \item Determine all positive integer solutions of the Diophantine equation $4x + 7y = 97$. \ispace
    \item Prove that the fraction $\displaystyle{\frac{21n + 4}{14n + 3}}$ is irreducible for every positive integer $n$. \ispace
\end{enumerate}

\section*{Advanced Problems}
\begin{enumerate}
    \item Determine $\gcd{(4 + 17i, 7 + 6i)}$. \aspace
    \item Determine the least positive integer $n$ satisfying $83 \equiv 75n \pmod{32}$. \aspace
\end{enumerate}
