\section*{Basic Problems}
\begin{enumerate}
    \item Determine the sum of $200 - 199 + 198 - 197 + 196 + \dots + 2 - 1$. \bspace
    
    \item Given that the sum of the first $n$ terms of a sequence is $n(n + 1)(n + 2)$, determine the 10th term of the sequence. \bspace
    
    \item Given that in an arithmetic sequence $t_p = q^2$ and $t_q = p^2$, determine the common difference. \bspace
\end{enumerate}

\section*{Intermediate Problems}
\begin{enumerate}
    \item The sum of the first $n$ terms in a sequence is $S_n = 33n - n^2$. Show that this is an arithmetic sequence. \ispace
    
    \item If $a, b, c,$ and $d$ form a geometric sequence, prove that $(a + b)^2, (b + c)^2, (c + d)^2$ also form a geometric sequence. \ispace
\end{enumerate}

\section*{Advanced Problems}
\begin{enumerate}
    \item Suppose a sequence ${a_n}$with $a_1 = 3$ satisfies $a_{n + 1} = a_n + 2(3)^n + 1$. Determine $a_n$. \newpage
    
    \item An arithmetic sequence ${a_n}$ satisfies $b_n = (\frac{1}{2})^{a_n}$. Given that $b_1 + b_2 + b_3 = \frac{21}{8}$ and $b_1b_2b_3 = \frac{1}{8}$, determine $a_n$.
\end{enumerate}
