\newcommand{\bspace}{\\ \\ \\ \\ \\ \\ \\} 
\section*{Basic Problems}
\begin{enumerate}
    \item Express $1 + 2 + 3 + 4 + 5 + \dots + n$ in sigma notation. \bspace
    \item Express $1 + 3 + 7 + 15 + 31 + 63 + 127 + 255 + 511 + 1023$ in sigma notation. \bspace
    \item In an arithmetic sequence $a_1 = 83$ and $a_4 = 98$, determine $a_n$. \bspace
    \item In a geometric sequence $a_5 = 32$ and $r = \frac{1}{2}$, determine $a_n$. \bspace
\end{enumerate}


\newcommand{\ispace}{\\ \\ \\ \\ \\ \\ \\ \\ \\ \\ \\ \\ \\ \\}
\section*{Intermediate Problems}
\begin{enumerate}
    \item Determine the sum of all even numbers between $1001$ and $1801$. \ispace
    \item In a geometric sequence, $a_2 = 2,$ and $r > 0$. If $a_3a_9 = 2a^2_5$, what is the value of $a_1$? \ispace
    \item In an arithmetic sequence, the sum of the first $n$ terms of an arithmetic sequence is $S_n = 4n^2 - n + 2$, what is $a_n$? \ispace
\end{enumerate}

\section*{Advanced Problems}
\begin{enumerate}
    \item Determine the sum of the following sequence:
    \begin{align*}
        S & = \frac{1}{2} + \frac{1}{6} + \frac{1}{12} + \frac{1}{20} + \dots + \frac{1}{10100}
    \end{align*} \newpage
    \item Define $f(x)=\frac{x^2}{1+x^2}$ and suppose that \begin{align*}
        A & = f(1) + f(2) + f(3) + \dots + f(2018) \\
        B & = f(1) + f(\frac{1}{2}) + f(\frac{1}{3}) + \dots + f(\frac{1}{2018})
    \end{align*}
    Determine the value of $A + B$
\end{enumerate}
