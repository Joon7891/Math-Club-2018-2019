\section*{Basic Problems}
\begin{enumerate}
    \item The roots of $x^2 + px + q = 0$ are $5$ and $-2$. What are the values of $p$ and $q$? \begSpace
    
    \item Let $m$ and $n$ be the two roots of the equation $x^2 - 15x + 28 = 0$. Find $(m + 1)(n + 1)$ \begSpace
\end{enumerate}
\section*{Intermediate Problems}
\begin{enumerate}
    \item The roots for the equation $x^2 + 4x - 5 = 0$ are also the roots of the equation $2x^3 + 9x^2 - 6x - 5 = 0$. What is the third root of the second equation? \newpage
    
    \item Let $g(x) = x^3 - 5x^2 + 2x - 7$, and let the roots of $g(x)$ be $p, q,$ and $r$. Compute $p^2qr + pq^2r + pqr^2$ \interSpace
\end{enumerate}
\section*{Advanced Problems}
\begin{enumerate}
    \item Let $a$ and $b$ be the roots of the equation $x - mx + 2 = 0$. Suppose that $a + \frac{1}{b}$  and $b + \frac{1}{a}$ are the roots of the equation $x^2 - px + q = 0$. What is the $q$? \advSpace
    
    \item If the sums of the reciprocals of the roots of the quadratic $3x^2 + 7x + k = 0$ is $\frac{7}{3}$, what is $k$?
\end{enumerate}
